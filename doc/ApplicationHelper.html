def camel_case_safe(text)
        return text.gsub(/[,.?!#<>\\/]/, '').split(' ').each { |s| s.capitalize! }.join.to_s
end

    id_safe(text) click to toggle source

    Makes a string safe for use in the 'ID' attribute 1st Removes artifact characters 2nd Removes multiple '-' characters

    # File app/helpers/application_helper.rb, line 24
def id_safe(text)
        text.gsub(/[ .?!]/, ' ' => '-', '.' => '', '?' => '', '!' => '').gsub(/-{2,}/, '-').downcase.to_s
end
